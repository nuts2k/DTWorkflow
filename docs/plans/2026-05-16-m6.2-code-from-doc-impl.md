```go
package validation

import (
	"fmt"
	"strings"
)

// CodeFromDocPath 校验 code_from_doc 任务的 doc_path 字段。
// API handler / 服务端 CLI / dtw CLI 三处共用。
func CodeFromDocPath(docPath string) error {
	if strings.TrimSpace(docPath) == "" {
		return fmt.Errorf("doc_path 不能为空")
	}
	// \ -> / 归一化
	normalized := strings.ReplaceAll(docPath, `\`, "/")
	// 禁止绝对路径
	if strings.HasPrefix(normalized, "/") {
		return fmt.Errorf("不能为绝对路径: %q", docPath)
	}
	// 禁止路径遍历
	for _, seg := range strings.Split(normalized, "/") {
		if seg == ".." {
			return fmt.Errorf("不能包含路径遍历 (..): %q", docPath)
		}
	}
	return nil
}
```

```go
// ResolveCodeFromDocConfig 解析指定仓库的最终 code_from_doc 配置。
func (c *Config) ResolveCodeFromDocConfig(repoFullName string) CodeFromDocConfig {
	if c == nil {
		return CodeFromDocConfig{}
	}
	merged := c.CodeFromDoc
	for _, repo := range c.Repos {
		if repo.Name != repoFullName || repo.CodeFromDoc == nil {
			continue
		}
		if repo.CodeFromDoc.Enabled != nil {
			merged.Enabled = repo.CodeFromDoc.Enabled
		}
		if repo.CodeFromDoc.AutoIterate != nil {
			merged.AutoIterate = repo.CodeFromDoc.AutoIterate
		}
		if repo.CodeFromDoc.ReviewOnFailure != nil {
			merged.ReviewOnFailure = repo.CodeFromDoc.ReviewOnFailure
		}
		break
	}
	return merged
}
```

```go
// ParseCodeFromDocOutput 双层 JSON 解析：外层 CLI 信封 → 内层 CodeFromDocOutput。
func ParseCodeFromDocOutput(raw string) (*CodeFromDocOutput, *model.CLIMeta, error) {
	var cli cliResponse
	if err := json.Unmarshal([]byte(raw), &cli); err != nil {
		return nil, nil, fmt.Errorf("CLI JSON 解析失败: %w", err)
	}
	meta := &model.CLIMeta{
		CostUSD:    cli.effectiveCostUSD(),
		DurationMs: cli.DurationMs,
		IsError:    cli.isExecutionError(),
		NumTurns:   cli.NumTurns,
		SessionID:  cli.SessionID,
	}
	if cli.isExecutionError() {
		return nil, meta, fmt.Errorf("Claude CLI 报告错误: type=%s, subtype=%s", cli.Type, cli.Subtype)
	}

	jsonText := extractJSON(cli.Result)
	var output CodeFromDocOutput
	if err := json.Unmarshal([]byte(jsonText), &output); err != nil {
		return nil, meta, fmt.Errorf("%w: %v", ErrCodeFromDocParseFailure, err)
	}

	if err := validateSuccessInvariant(&output); err != nil {
		return nil, meta, fmt.Errorf("%w: %v", ErrCodeFromDocParseFailure, err)
	}

	return &output, meta, nil
}

// validateSuccessInvariant 成功不变量：Success=true 时 BranchName 和 CommitSHA 必须非空，FailureCategory 必须为 none。
func validateSuccessInvariant(o *CodeFromDocOutput) error {
	if o == nil || !o.Success {
		return nil
	}
	if o.BranchName == "" || o.CommitSHA == "" {
		return fmt.Errorf("不变量违反：success=true 但 branch_name=%q commit_sha=%q", o.BranchName, o.CommitSHA)
	}
	if o.FailureCategory != FailureCategoryNone {
		return fmt.Errorf("不变量违反：success=true 但 failure_category=%q", o.FailureCategory)
	}
	return nil
}

// extractJSON 从 CLI result 字段中提取 JSON 对象。
// result 可能含有前导/尾随非 JSON 文本（Claude 的 conversational wrapper）。
func extractJSON(s string) string {
	start := strings.Index(s, "{")
	if start < 0 {
		return s
	}
	end := strings.LastIndex(s, "}")
	if end < start {
		return s
	}
	return s[start : end+1]
}

var consecutiveDash = regexp.MustCompile(`-{2,}`)

// DocSlug 从 doc_path 派生分支后缀。
func DocSlug(docPath string) string {
	if docPath == "" {
		return "unnamed"
	}
	base := filepath.Base(docPath)
	ext := filepath.Ext(base)
	name := strings.TrimSuffix(base, ext)

	slug := docSlugPattern.ReplaceAllString(name, "-")
	slug = consecutiveDash.ReplaceAllString(slug, "-")
	slug = strings.Trim(slug, "-")

	if len(slug) > 50 {
		slug = slug[:50]
		slug = strings.TrimRight(slug, "-")
	}
	if slug == "" {
		return "unnamed"
	}
	return slug
}

	for _, f := range files {
		switch f.Action {
		case "created":
			created++
		case "modified":
			modified++
		}
	}
	return
}
```

```go
// Execute 主编排入口。
func (s *Service) Execute(ctx context.Context, payload model.TaskPayload) (*CodeFromDocResult, error) {
	owner, repo := payload.RepoOwner, payload.RepoName
	result := &CodeFromDocResult{}

	// 1. 校验
	if payload.DocPath == "" {
		return nil, fmt.Errorf("doc_path 为空: %w", ErrInfoInsufficient)
	}

	// 2. 配置
	cfg := s.cfgProv.ResolveCodeFromDocConfig(payload.RepoFullName)
	maxRetry := cfg.MaxRetryRounds
	if maxRetry <= 0 {
		maxRetry = 3
	}

	// 3. 确定分支
	branch := payload.HeadRef
	isPathA := branch == "" || branch == payload.BaseRef
	if isPathA {
		branch = "auto-code/" + payload.DocSlug
	}

	// 4. 构造 prompt
	prompt := BuildCodeFromDocPrompt(PromptContext{
		Owner:          owner,
		Repo:           repo,
		Branch:         branch,
		BaseRef:        payload.BaseRef,
		DocPath:        payload.DocPath,
		MaxRetryRounds: maxRetry,
	})

	// 5. 容器执行
	cmd := []string{"claude", "-p", "--output-format", "json", "-"}
	execResult, err := s.pool.RunWithCommandAndStdin(ctx, payload, cmd, []byte(prompt))
	if err != nil {
		result.RawOutput = safeOutput(execResult)
		return result, fmt.Errorf("容器执行失败: %w", err)
	}
	if execResult == nil {
		return result, fmt.Errorf("容器执行结果为空")
	}
	result.RawOutput = execResult.Output
	result.ExitCode = execResult.ExitCode

	if execResult.ExitCode != 0 {
		s.logger.ErrorContext(ctx, "code_from_doc worker 非零退出",
			"exit_code", execResult.ExitCode)
		return result, nil
	}

	// 6. 解析
	output, meta, parseErr := ParseCodeFromDocOutput(execResult.Output)
	result.CLIMeta = meta
	if parseErr != nil {
		result.ParseError = parseErr
		s.logger.ErrorContext(ctx, "code_from_doc 结果解析失败",
			"parse_error", parseErr)
		return result, fmt.Errorf("%w", ErrCodeFromDocParseFailure)
	}
	result.Output = output
	sanitizeCodeFromDocOutput(output)

	// 7. 分支路由：按 failure_category 分发
	switch output.FailureCategory {
	case FailureCategoryInfoInsufficient:
		s.persistResult(ctx, payload, result, branch)
		return result, fmt.Errorf("%w", ErrInfoInsufficient)

	case FailureCategoryInfrastructure:
		return result, fmt.Errorf("基础设施故障: %s", output.FailureReason)

	case FailureCategoryTestFailure:
		// 仍创建 PR 保留成果，但不添加 auto-iterate 标签
		prNum, prURL, prErr := s.createOrFindPR(ctx, payload, output, branch, isPathA)
		if prErr != nil {
			s.logger.ErrorContext(ctx, "test_failure 场景创建 PR 失败",
				"error", prErr)
		} else {
			result.PRNumber = prNum
			result.PRURL = prURL
		}
		s.persistResult(ctx, payload, result, branch)
		s.maybeEnqueueReview(ctx, payload, result, cfg, branch)
		return result, fmt.Errorf("%w", ErrTestFailure)

	case FailureCategoryNone:
		if !output.Success {
			s.persistResult(ctx, payload, result, branch)
			return result, fmt.Errorf("Claude 返回 success=false（category=none）: %w", ErrCodeFromDocParseFailure)
		}
	}

	// 8. Success=true → 创建 PR
	prNum, prURL, prErr := s.createOrFindPR(ctx, payload, output, branch, isPathA)
	if prErr != nil {
		s.logger.ErrorContext(ctx, "创建 PR 失败",
			"branch", branch, "error", prErr)
		s.persistResult(ctx, payload, result, branch)
		return result, fmt.Errorf("创建 PR 失败: %w", prErr)
	}
	result.PRNumber = prNum
	result.PRURL = prURL

	// 9. auto-iterate 标签
	if cfg.AutoIterate && s.labelClient != nil {
		s.addAutoIterateLabel(ctx, owner, repo, prNum)
	}

	// 10. 持久化
	s.persistResult(ctx, payload, result, branch)

	// 11. 主动入队 review
	s.maybeEnqueueReview(ctx, payload, result, cfg, branch)

	return result, nil
}

func (s *Service) createOrFindPR(ctx context.Context, payload model.TaskPayload,
	output *CodeFromDocOutput, branch string, isPathA bool) (int64, string, error) {
	if s.prClient == nil {
		return 0, "", fmt.Errorf("PRClient 未注入")
	}

	// 路径 B：检查是否已有 open PR
	if !isPathA {
		if num, url, ok := s.findExistingPR(ctx, payload.RepoOwner, payload.RepoName, branch); ok {
			s.logger.InfoContext(ctx, "路径 B：分支已有 open PR，跳过创建",
				"branch", branch, "pr_number", num)
			return num, url, nil
		}
	}

	// 幂等检查（路径 A 重试场景）
	if num, url, ok := s.findExistingPR(ctx, payload.RepoOwner, payload.RepoName, branch); ok {
		s.logger.InfoContext(ctx, "复用已存在的 PR（幂等）",
			"branch", branch, "pr_number", num)
		return num, url, nil
	}

	// 确定 base
	base := payload.BaseRef
	if base == "" {
		repoInfo, _, err := s.prClient.GetRepo(ctx, payload.RepoOwner, payload.RepoName)
		if err != nil {
			return 0, "", fmt.Errorf("获取仓库默认分支失败: %w", err)
		}
		base = repoInfo.DefaultBranch
	}

	title := fmt.Sprintf("feat: 自动实现 %s", payload.DocPath)
	title = sanitizeGiteaText(title)
	title = truncateByRune(title, 240)

	body := formatCodeFromDocPRBody(output, payload.DocPath)

	// 5xx 退避重试
	var (
		pr         *gitea.PullRequest
		err        error
		lastStatus int
		totalTries = len(prCreateBackoffs) + 1
	)
	for attempt := 0; attempt < totalTries; attempt++ {
		if attempt > 0 {
			backoff := prCreateBackoffs[attempt-1]
			s.logger.WarnContext(ctx, "创建 PR 失败，按退避等待重试",
				"attempt", attempt, "backoff", backoff, "error", err)
			select {
			case <-ctx.Done():
				return 0, "", fmt.Errorf("等待重试时 ctx 取消: %w", ctx.Err())
			case <-time.After(backoff):
			}
		}
		pr, _, err = s.prClient.CreatePullRequest(ctx, payload.RepoOwner, payload.RepoName,
			gitea.CreatePullRequestOption{
				Head:  branch,
				Base:  base,
				Title: title,
				Body:  body,
			})
		if err == nil {
			break
		}
		lastStatus = extractStatusCode(err)
		if num, url, ok := s.findExistingPR(ctx, payload.RepoOwner, payload.RepoName, branch); ok {
			return num, url, nil
		}
		if lastStatus < 500 {
			break
		}
	}
	if err != nil {
		return 0, "", err
	}
	if pr == nil {
		return 0, "", fmt.Errorf("Gitea API 返回空 PR")
	}
	s.logger.InfoContext(ctx, "PR 已创建",
		"branch", branch, "pr_number", pr.Number, "pr_url", pr.HTMLURL)
	return pr.Number, pr.HTMLURL, nil
}

func (s *Service) findExistingPR(ctx context.Context, owner, repo, headBranch string) (int64, string, bool) {
	if headBranch == "" || s.prClient == nil {
		return 0, "", false
	}
	prs, _, err := gitea.PaginateAll(ctx, 50, 10,
		func(ctx context.Context, page, pageSize int) ([]*gitea.PullRequest, *gitea.Response, error) {
			return s.prClient.ListRepoPullRequests(ctx, owner, repo, gitea.ListPullRequestsOptions{
				State:       "open",
				ListOptions: gitea.ListOptions{Page: page, PageSize: pageSize},
			})
		})
	if err != nil {
		s.logger.WarnContext(ctx, "查询既有 PR 失败", "error", err)
		return 0, "", false
	}
	for _, pr := range prs {
		if pr != nil && pr.Head != nil && pr.Head.Ref == headBranch {
			return pr.Number, pr.HTMLURL, true
		}
	}
	return 0, "", false
}

}

func (s *Service) persistResult(ctx context.Context, payload model.TaskPayload, result *CodeFromDocResult, branch string) {

		return

	testPassed, testFailed := 0, 0

		failReason = result.Output.FailureReason
		impl = result.Output.Implementation
	}
	rec := &store.CodeFromDocResultRecord{

		Repo:            payload.RepoFullName,

		DocPath:         payload.DocPath,
		Success:         result.Output != nil && result.Output.Success,

		FailureCategory: failCat,
		FailureReason:   failReason,
		FilesCreated:    created,

		TestFailed:      testFailed,

	if err := s.store.SaveCodeFromDocResult(ctx, rec); err != nil {

	}

func (s *Service) maybeEnqueueReview(ctx context.Context, payload model.TaskPayload,
	result *CodeFromDocResult, cfg CodeFromDocConfigView, branch string) {
	if s.reviewEnqueuer == nil || result.PRNumber <= 0 {
		return
	}
	shouldEnqueue := (result.Output != nil && result.Output.Success) || cfg.ReviewOnFailure
	if !shouldEnqueue {
		return
	}
	reviewPayload := model.TaskPayload{
		TaskType:     model.TaskTypeReviewPR,
		RepoOwner:    payload.RepoOwner,
		RepoName:     payload.RepoName,
		RepoFullName: payload.RepoFullName,
		CloneURL:     payload.CloneURL,
		PRNumber:     result.PRNumber,
		HeadRef:      branch,
		BaseRef:      payload.BaseRef,
	}
	triggeredBy := fmt.Sprintf("code_from_doc:%s", payload.TaskID)
	_, err := s.reviewEnqueuer.EnqueueManualReview(ctx, reviewPayload, triggeredBy)
	if err != nil {
		s.logger.WarnContext(ctx, "主动入队 review 失败", "error", err)
		return
	}
	if s.store != nil && payload.TaskID != "" {
		if err := s.store.UpdateCodeFromDocReviewEnqueued(ctx, payload.TaskID); err != nil {
			s.logger.WarnContext(ctx, "更新 review_enqueued 失败", "error", err)
		}
	}
}

func formatCodeFromDocPRBody(output *CodeFromDocOutput, docPath string) string {
	var b strings.Builder
	b.WriteString("## 自动编码\n\n")
	b.WriteString(fmt.Sprintf("基于设计文档 `%s` 自动实现。\n\n", docPath))

	if output != nil {
		if output.Implementation != "" {
			b.WriteString("### 实现概要\n\n")
			b.WriteString(output.Implementation)
			b.WriteString("\n\n")
		}
		if len(output.ModifiedFiles) > 0 {
			b.WriteString("### 变更文件\n\n")
			for _, f := range output.ModifiedFiles {
				b.WriteString(fmt.Sprintf("- `%s` (%s)\n", f.Path, f.Action))
			}
			b.WriteString("\n")
		}
		b.WriteString(fmt.Sprintf("### 测试结果\n\n- 通过: %d\n- 失败: %d\n- 跳过: %d\n",
			output.TestResults.Passed, output.TestResults.Failed, output.TestResults.Skipped))
	}
	return sanitizeGiteaText(b.String())
}

func safeOutput(r *worker.ExecutionResult) string {
	if r == nil {
		return ""
	}
	return r.Output
}

// sanitizeGiteaText 过滤 U+10000 以上字符（Gitea MySQL utf8 兼容）。
func sanitizeGiteaText(s string) string {
	var b strings.Builder
	b.Grow(len(s))
	for _, r := range s {
		if r > 0xFFFF {
			b.WriteRune(0xFFFD)
		} else {
			b.WriteRune(r)
		}
	}
	return b.String()
}

func truncateByRune(s string, maxRunes int) string {
	if maxRunes <= 0 {
		return ""
	}
	if utf8.RuneCountInString(s) <= maxRunes {
		return s
	}
	runes := []rune(s)
	return string(runes[:maxRunes])
}

func extractStatusCode(err error) int {
	var errResp *gitea.ErrorResponse
	if errors.As(err, &errResp) {
		return errResp.StatusCode
	}
	return 0
}
```

```go
// Cancel-and-Replace 按 (repo, branch) 聚合——同一分支同时只能有一个 code_from_doc 任务。
func (h *EnqueueHandler) EnqueueCodeFromDoc(ctx context.Context, payload model.TaskPayload, triggeredBy string) (string, error) {
	if payload.RepoFullName == "" || payload.CloneURL == "" {
		return "", fmt.Errorf("payload 数据不完整: RepoFullName 或 CloneURL 为空")
	}
	if payload.DocPath == "" {
		return "", fmt.Errorf("doc_path 不能为空")
	}

	payload.TaskType = model.TaskTypeCodeFromDoc
	// 幂等 DeliveryID：manual:code_from_doc:{repo}:{branch}:{doc_path_hash}
	branch := payload.HeadRef
	if branch == "" {
		branch = "auto-code/" + payload.DocSlug
	}
	payload.DeliveryID = fmt.Sprintf("manual:code_from_doc:%s:%s:%x",
		payload.RepoFullName, branch, fnvHash(payload.DocPath))

	// 幂等检查
	existing, err := h.store.FindByDeliveryID(ctx, payload.DeliveryID, model.TaskTypeCodeFromDoc)
	if err != nil {
		return "", fmt.Errorf("幂等检查失败: %w", err)
	}
	if existing != nil {
		h.logger.InfoContext(ctx, "code_from_doc 任务已存在,跳过",
			"delivery_id", payload.DeliveryID,
			"task_id", existing.ID,
			"status", existing.Status)
		return existing.ID, nil
	}

	// Cancel-and-Replace：按 (repo, branch) 聚合
	activeTasks := h.listActiveCodeFromDocTasks(ctx, payload.RepoFullName, branch)

	record := &model.TaskRecord{
		TaskType:     model.TaskTypeCodeFromDoc,
		Priority:     model.PriorityNormal,
		RepoFullName: payload.RepoFullName,
		DeliveryID:   payload.DeliveryID,
		TriggeredBy:  triggeredBy,
	}

	if err := h.enqueueTask(ctx, payload, record); err != nil {
		return "", err
	}
	h.cancelTasks(ctx, activeTasks)

	h.logger.InfoContext(ctx, "code_from_doc 任务已入队",
		"task_id", record.ID, "repo", payload.RepoFullName,
		"doc_path", payload.DocPath, "branch", branch,
		"triggered_by", triggeredBy)
	return record.ID, nil
}

// listActiveCodeFromDocTasks 查找同仓库同分支的活跃 code_from_doc 任务。
func (h *EnqueueHandler) listActiveCodeFromDocTasks(ctx context.Context, repoFullName, branch string) []*model.TaskRecord {

		}

// fnvHash 简易 FNV-1a 哈希，用于 DeliveryID 中的 doc_path 摘要。
func fnvHash(s string) uint32 {
	const (
		offset32 = 2166136261
		prime32  = 16777619
	)
	h := uint32(offset32)
	for i := 0; i < len(s); i++ {
		h ^= uint32(s[i])
		h *= prime32
```

```go
package api

import (

)

type codeFromDocRequest struct {
	DocPath string `json:"doc_path"`
	Branch  string `json:"branch,omitempty"`
	Ref     string `json:"ref,omitempty"`
}

const codeFromDocMaxBodyBytes = 1 << 14

func (h *handlers) triggerCodeFromDoc(c *gin.Context) {
	owner := c.Param("owner")
	repo := c.Param("repo")
	if owner == "" || repo == "" {
		Error(c, http.StatusBadRequest, ErrCodeBadRequest, "owner 和 repo 不能为空")
		return
	}

	var req codeFromDocRequest
	if c.Request.Body != nil {
		limited := http.MaxBytesReader(c.Writer, c.Request.Body, codeFromDocMaxBodyBytes)
		body, err := io.ReadAll(limited)
		if err != nil {
			if strings.Contains(err.Error(), "request body too large") {
				Error(c, http.StatusRequestEntityTooLarge, ErrCodeBadRequest,
					fmt.Sprintf("请求体过大（上限 %d 字节）", codeFromDocMaxBodyBytes))
				return
			}
			Error(c, http.StatusBadRequest, ErrCodeBadRequest, "读取请求体失败: "+err.Error())
			return
		}
		if len(bytes.TrimSpace(body)) > 0 {
			if err := json.Unmarshal(body, &req); err != nil {
				Error(c, http.StatusBadRequest, ErrCodeBadRequest, "无效的请求体: "+err.Error())
				return
			}
		}
	}

	if req.DocPath == "" {
		Error(c, http.StatusBadRequest, ErrCodeBadRequest, "doc_path 不能为空")
		return
	}
	if err := validation.CodeFromDocPath(req.DocPath); err != nil {
		Error(c, http.StatusBadRequest, ErrCodeBadRequest, "doc_path "+err.Error())
		return
	}

	if h.deps.GiteaClient == nil {
		Error(c, http.StatusBadGateway, ErrCodeBadGateway, "Gitea 客户端未配置")
		return
	}
	if h.deps.EnqueueHandler == nil {
		Error(c, http.StatusInternalServerError, ErrCodeInternalError, "入队服务未配置")
		return
	}

	ctx := c.Request.Context()

	repoInfo, _, err := h.deps.GiteaClient.GetRepo(ctx, owner, repo)
	if err != nil {
		if gitea.IsNotFound(err) {
			Error(c, http.StatusNotFound, ErrCodeNotFound,
				fmt.Sprintf("仓库 %s/%s 不存在", owner, repo))
			return
		}
		h.deps.Logger.Error("查询 Gitea 仓库失败", "owner", owner, "repo", repo, "error", err)
		Error(c, http.StatusBadGateway, ErrCodeBadGateway, "查询 Gitea 仓库信息失败")
		return
	}

	identity, _ := c.Get(ContextKeyIdentity)
	triggeredBy := fmt.Sprintf("manual:%v", identity)

	baseRef := req.Ref
	if baseRef == "" {
		baseRef = repoInfo.DefaultBranch
	}

	docSlug := code.DocSlug(req.DocPath)

	payload := model.TaskPayload{
		TaskType:     model.TaskTypeCodeFromDoc,
		RepoOwner:    owner,
		RepoName:     repo,
		RepoFullName: repoInfo.FullName,
		CloneURL:     repoInfo.CloneURL,
		DocPath:      req.DocPath,
		DocSlug:      docSlug,
		BaseRef:      baseRef,
		HeadRef:      req.Branch,
	}

	taskID, err := h.deps.EnqueueHandler.EnqueueCodeFromDoc(ctx, payload, triggeredBy)
	if err != nil {
		h.deps.Logger.Error("入队 code_from_doc 失败",
			"owner", owner, "repo", repo, "doc_path", req.DocPath, "error", err)
		Error(c, http.StatusInternalServerError, ErrCodeInternalError, "入队 code_from_doc 失败")
		return
	}

	Success(c, http.StatusAccepted, gin.H{
		"task_id": taskID,
		"message": fmt.Sprintf("code_from_doc 任务已入队（doc=%q, branch=%q）", req.DocPath, req.Branch),
	})
}
```

```go
			printer.PrintHuman("code_from_doc 任务已创建: %s", taskID)
		}

		opts := dtw.DefaultWaitOptions()
		if cfdTimeout > 0 {
			opts.Timeout = cfdTimeout
		} else {
			opts.Timeout = 65 * time.Minute // 略大于默认 60m 容器超时
		}

		if !flagJSON {
			printer.PrintHuman("等待任务完成...")
		}
		status, err := dtw.WaitForTask(cmd.Context(), client, taskID, opts)
		if err != nil {
			return fmt.Errorf("等待任务失败: %w", err)
		}

		if flagJSON {
			return printer.PrintJSON(status)
		}

		printer.PrintHuman("任务状态: %s", status.Status)
		if status.Error != "" {
			printer.PrintHuman("错误: %s", status.Error)
		}
		if status.Status == "failed" {
			return fmt.Errorf("code_from_doc 任务失败")
		}
		return nil
	},
}
```

```go
func (a *configAdapter) ResolveCodeFromDocConfig(repoFullName string) codesvc.CodeFromDocConfigView {

		MaxRetryRounds:  resolved.MaxRetryRounds,
		ReviewOnFailure: resolved.ReviewOnFailure != nil && *resolved.ReviewOnFailure,

}
```
