> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** 在当前 M1.5-M1.7 已完成的任务执行与通知基线上，引入统一的 Viper 配置管理能力，使 `root`、`serve`、`task` 共用 YAML/环境变量/CLI flags 配置入口，并将通知路由迁移为配置驱动。

package config

import (
    "fmt"
    "sync"
    "time"

    "github.com/spf13/viper"
)

type Config struct {
    Server   ServerConfig   `mapstructure:"server"`
    Gitea    GiteaConfig    `mapstructure:"gitea"`
    Claude   ClaudeConfig   `mapstructure:"claude"`
    Redis    RedisConfig    `mapstructure:"redis"`
    Database DatabaseConfig `mapstructure:"database"`
    Log      LogConfig      `mapstructure:"log"`
    Worker   WorkerConfig   `mapstructure:"worker"`
    Webhook  WebhookConfig  `mapstructure:"webhook"`
    Notify   NotifyConfig   `mapstructure:"notify"`
    Repos    []RepoConfig   `mapstructure:"repos"`
}

type ServerConfig struct {
    Host string `mapstructure:"host"`
    Port int    `mapstructure:"port"`
}

type GiteaConfig struct {
    URL                string `mapstructure:"url"`
    Token              string `mapstructure:"token"`
    InsecureSkipVerify bool   `mapstructure:"insecure_skip_verify"`
}

type ClaudeConfig struct {
    APIKey string `mapstructure:"api_key"`
}

type RedisConfig struct {
    Addr     string `mapstructure:"addr"`
    Password string `mapstructure:"password"`
    DB       int    `mapstructure:"db"`
}

type DatabaseConfig struct {
    Path string `mapstructure:"path"`
}

type LogConfig struct {
    Level  string `mapstructure:"level"`
    Format string `mapstructure:"format"`
}

type WorkerConfig struct {
    Concurrency int           `mapstructure:"concurrency"`
    Timeout     time.Duration `mapstructure:"timeout"`
    Image       string        `mapstructure:"image"`
    CPULimit    string        `mapstructure:"cpu_limit"`
    MemoryLimit string        `mapstructure:"memory_limit"`
    NetworkName string        `mapstructure:"network_name"`
}

type WebhookConfig struct {
    Secret string `mapstructure:"secret"`
}

type NotifyConfig struct {
    DefaultChannel string                   `mapstructure:"default_channel"`
    Channels       map[string]ChannelConfig `mapstructure:"channels"`
    Routes         []RouteConfig            `mapstructure:"routes"`
}

type ChannelConfig struct {
    Enabled bool              `mapstructure:"enabled"`
    Options map[string]string `mapstructure:"options"`
}

type RouteConfig struct {
    Repo     string   `mapstructure:"repo"`
    Events   []string `mapstructure:"events"`
    Channels []string `mapstructure:"channels"`
}

type RepoConfig struct {
    Name   string          `mapstructure:"name"`
    Review *ReviewOverride `mapstructure:"review,omitempty"`
    Notify *NotifyOverride `mapstructure:"notify,omitempty"`
}

type ReviewOverride struct {
    Enabled        *bool    `mapstructure:"enabled,omitempty"`
    IgnorePatterns []string `mapstructure:"ignore_patterns,omitempty"`
    Severity       string   `mapstructure:"severity,omitempty"`
}

type NotifyOverride struct {
    Routes []RouteConfig `mapstructure:"routes,omitempty"`
}

type Manager struct {
    v       *viper.Viper
    mu      sync.RWMutex
    current *Config
}

type ManagerOption func(*Manager) error

func NewManager(opts ...ManagerOption) (*Manager, error) {
    m := &Manager{v: viper.New()}
    for _, opt := range opts {
        if err := opt(m); err != nil {
            return nil, err
        }
    }
    return m, nil
}

func WithConfigFile(path string) ManagerOption {
    return func(m *Manager) error {
        m.v.SetConfigFile(path)
        return nil
    }
}

func WithEnvPrefix(prefix string) ManagerOption {
    return func(m *Manager) error {
        m.v.SetEnvPrefix(prefix)
        m.v.SetEnvKeyReplacer(strings.NewReplacer(".", "_"))
        m.v.AutomaticEnv()
        return nil
    }
}

func WithDefaults() ManagerOption {
    return func(m *Manager) error {
        m.v.SetDefault("server.host", "0.0.0.0")
        m.v.SetDefault("server.port", 8080)
        m.v.SetDefault("log.level", "info")
        m.v.SetDefault("log.format", "text")
        m.v.SetDefault("worker.concurrency", 3)
        m.v.SetDefault("worker.timeout", "30m")
        m.v.SetDefault("worker.image", "dtworkflow-worker:1.0")
        m.v.SetDefault("worker.cpu_limit", "2.0")
        m.v.SetDefault("worker.memory_limit", "4g")
        m.v.SetDefault("worker.network_name", "dtworkflow-net")
        m.v.SetDefault("database.path", "data/dtworkflow.db")
        m.v.SetDefault("redis.addr", "localhost:6379")
        m.v.SetDefault("redis.db", 0)
        m.v.SetDefault("notify.default_channel", "gitea")
        return nil
    }
}

func (m *Manager) Load() error {
    if err := m.v.ReadInConfig(); err != nil {
        return fmt.Errorf("读取配置文件: %w", err)
    }
    cfg := &Config{}
    if err := m.v.Unmarshal(cfg); err != nil {
        return fmt.Errorf("解析配置: %w", err)
    }
    m.mu.Lock()
    m.current = cfg
    m.mu.Unlock()
    return nil
}

func (m *Manager) Get() *Config {
    m.mu.RLock()
    defer m.mu.RUnlock()
    return m.current
}

func (m *Manager) Viper() *viper.Viper { return m.v }
```

package config

import (
    "fmt"
    "strings"
)

type ValidationError struct {
    Errors []error
}

func (e *ValidationError) Error() string {
    parts := make([]string, 0, len(e.Errors))
    for _, err := range e.Errors {
        parts = append(parts, err.Error())
    }
    return strings.Join(parts, "; ")
}

func Validate(cfg *Config) error {
    var errs []error
    if cfg == nil {
        return fmt.Errorf("config 不能为空: %w", ErrInvalidConfig)
    }
    if cfg.Gitea.URL == "" {
        errs = append(errs, fmt.Errorf("gitea.url 不能为空"))
    }
    if cfg.Gitea.Token == "" {
        errs = append(errs, fmt.Errorf("gitea.token 不能为空"))
    }
    if cfg.Claude.APIKey == "" {
        errs = append(errs, fmt.Errorf("claude.api_key 不能为空"))
    }
    if cfg.Webhook.Secret == "" {
        errs = append(errs, fmt.Errorf("webhook.secret 不能为空"))
    }
    if cfg.Server.Port < 1 || cfg.Server.Port > 65535 {
        errs = append(errs, fmt.Errorf("server.port 必须在 1-65535 之间"))
    }
    if cfg.Worker.Concurrency < 1 {
        errs = append(errs, fmt.Errorf("worker.concurrency 必须 >= 1"))
    }
    if cfg.Notify.DefaultChannel != "" {
        ch, ok := cfg.Notify.Channels[cfg.Notify.DefaultChannel]
        if !ok || !ch.Enabled {
            errs = append(errs, fmt.Errorf("notify.default_channel %q 未配置或未启用", cfg.Notify.DefaultChannel))
        }
    }
    for i, route := range cfg.Notify.Routes {
        for _, ch := range route.Channels {
            if _, ok := cfg.Notify.Channels[ch]; !ok {
                errs = append(errs, fmt.Errorf("notify.routes[%d] 引用了未配置的渠道: %q", i, ch))
            }
        }
    }
    for i, repo := range cfg.Repos {
        if repo.Name == "" || !strings.Contains(repo.Name, "/") {
            errs = append(errs, fmt.Errorf("repos[%d].name 必须为 owner/repo 格式", i))
        }
    }
    if len(errs) > 0 {
        return &ValidationError{Errors: errs}
    }
    return nil
}
```

package config

func (c *Config) ResolveNotifyRoutes(repoFullName string) []RouteConfig {
    for _, repo := range c.Repos {
        if repo.Name == repoFullName && repo.Notify != nil && len(repo.Notify.Routes) > 0 {
            return append([]RouteConfig(nil), repo.Notify.Routes...)
        }
    }
    return append([]RouteConfig(nil), c.Notify.Routes...)
}

func (c *Config) ResolveReviewConfig(repoFullName string) ReviewOverride {
    var out ReviewOverride
    for _, repo := range c.Repos {
        if repo.Name == repoFullName && repo.Review != nil {
            if repo.Review.Enabled != nil {
                out.Enabled = repo.Review.Enabled
            }
            if len(repo.Review.IgnorePatterns) > 0 {
                out.IgnorePatterns = append([]string(nil), repo.Review.IgnorePatterns...)
            }
            if repo.Review.Severity != "" {
                out.Severity = repo.Review.Severity
            }
            return out
        }
    }
    return out
}

func buildNotifyRules(cfg *config.Config, repoFullName string) ([]notify.RoutingRule, string) {
    if cfg == nil {
        return nil, ""
    }
    routes := cfg.ResolveNotifyRoutes(repoFullName)
    rules := make([]notify.RoutingRule, 0, len(routes))
    for _, route := range routes {
        eventTypes := make([]notify.EventType, 0, len(route.Events))
        for _, event := range route.Events {
            eventTypes = append(eventTypes, notify.EventType(event))
        }
        rules = append(rules, notify.RoutingRule{
            RepoPattern: route.Repo,
            EventTypes:  eventTypes,
            Channels:    append([]string(nil), route.Channels...),
        })
    }
    return rules, cfg.Notify.DefaultChannel
}
```

将 `buildNotifier` 改为：

func buildNotifier(cfg *config.Config, giteaClient *gitea.Client) (queue.TaskNotifier, error) {
    if giteaClient == nil || cfg == nil {
        return nil, nil
    }

    giteaNotifier, err := notify.NewGiteaNotifier(&giteaCommentAdapter{client: giteaClient}, notify.WithLogger(slog.Default()))
    if err != nil {
        return nil, fmt.Errorf("构造 GiteaNotifier 失败: %w", err)
    }

    rules, fallback := buildNotifyRules(cfg, "")
    router, err := notify.NewRouter(
        notify.WithNotifier(giteaNotifier),
        notify.WithRules(rules),
        notify.WithFallback(fallback),
        notify.WithRouterLogger(slog.Default()),
    )
    if err != nil {
        return nil, fmt.Errorf("构造通知路由失败: %w", err)

}

package config

import (
    "fmt"
    "log/slog"

    "github.com/fsnotify/fsnotify"
)

func (m *Manager) WatchConfig() {
    m.v.OnConfigChange(func(_ fsnotify.Event) {
        cfg := &Config{}
        if err := m.v.Unmarshal(cfg); err != nil {
            slog.Error("配置热加载解析失败", "error", err)
            return
        }
        if err := Validate(cfg); err != nil {
            slog.Error("配置热加载校验失败", "error", err)
            return
        }

        m.mu.Lock()
        oldCfg := m.current
        m.current = cfg
        callbacks := append([]func(oldCfg, newCfg *Config){}, m.onChange...)
        m.mu.Unlock()

        for _, cb := range callbacks {
            cb(oldCfg, cfg)
        }
    })
    m.v.WatchConfig()
}
```

本轮只实现配置快照更新与回调通知，不在这里做 Redis/SQLite/Worker Pool 的在线重建。
